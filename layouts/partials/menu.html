{{- end -}} {{- define "partials/inline/menu/walk.html" -}} {{- $page := .page -}} {{- range .menuEntries -}} {{- $attrs := dict "href" .URL -}} {{- with .Params -}} {{- range $key, $value := . -}} {{- $attrs = merge $attrs (dict $key $value) -}} {{- end -}} {{- end -}} {{- if $page.IsMenuCurrent .Menu . -}} {{- $attrs = merge $attrs (dict "class" "active" "aria-current" "page") -}} {{- else if $page.HasMenuCurrent .Menu . -}} {{- $attrs = merge $attrs (dict "class" "ancestor" "aria-current" "true") -}} {{- end -}} {{- $name := .Name -}} {{- with .Identifier -}} {{- with T . -}} {{- $name = . -}} {{- end -}} {{- end }} {{ $name }} {{- end -}} {{- end -}}
